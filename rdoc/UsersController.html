def update

        if @user.update_attributes(user_params)

def update_role
        @user = User.find(params[:id])
        if @user.update_attribute(:role, params[:role])
                flash[:success] = "Role Changed"
        else
                flash[:error] = "Error"
        end
        redirect_to :back
end
